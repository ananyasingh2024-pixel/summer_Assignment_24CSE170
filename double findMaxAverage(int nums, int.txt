double findMaxAverage(int* nums, int numsSize, int k) {
 long long windowSum = 0;
 for (int i =0; i <k; i++) {
    windowSum += nums[i];
 }
 long long maxSum =windowSum;
 for (int i =k; i < numsSize;i++){
    windowSum += nums[i] - nums[i-k];
    if(windowSum > maxSum){
        maxSum = windowSum;
    }
 }
   return (double)maxSum / k; 
 }
